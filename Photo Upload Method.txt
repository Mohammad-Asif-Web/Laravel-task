*** make additional controller for photo upload and remove***
photo upload and remove onek page kora hobe, tai bar bar code na lekhe, er jnne akti controller baniye nite hobe. Controller create krar por sobar age 'Intervention Image' er duita Facades require krte hobe.
use Intervention\Image\ImageManager;
use Intervention\Image\Drivers\Gd\Driver;

class PhotoUploadController extends Controller
{
  public static function imageUpload($name, $height, $width, $path, $file)
  {
     $manager = new ImageManager(new Driver());
     // ata hocche full imager nam
     $image_name = $name.'.webp';
     $img = $manager->read($file);
     $img = $img->resize($width,$height)->save(public_path($path).$image_name, 50, 'webp');
     return $image_name;
  }

  public static function imageUnlink($path, $name)
  {
     $image_path = public_path($path).$name;
     if(file_exists($image_path)){
        unlink($image_path);
     }
  }
}

ai controller e parameter hisebe $name, $height, $width, $path and $file deya ache, jekono photo niye kaj krte ai data gulo lagei, tai deya hoyeche.
